pipeline {
    agent any
    tools{
        jdk 'jdk21'
        nodejs 'node20'
    }
    environment {
        SCANNER_HOME = tool 'sonar-scanner'
    }

    stages {
        stage('clean workspace') {
            steps {
                cleanWs()
            }
        }
        stage('Checkout') {
            steps {
               git branch: 'main', url: 'https://github.com/linuxdocs2025/robot-gallery.git'
            }
        }
        stage("Sonarqube Analysis "){
             steps{
                 withSonarQubeEnv('sonar-server') {
                     sh ''' $SCANNER_HOME/bin/sonar-scanner -Dsonar.projectName=robot-gallery \
                     -Dsonar.projectKey=robot-gallery '''
                 }
             }
         }
        stage('Install Dependencies') {
            steps {
                sh "npm install"
            }
        }
        stage('TRIVY FS SCAN') {
             steps {
                 sh "trivy fs . > trivyfs.txt"
             }
         }
        stage("Docker Build & Push"){
             steps{
                 script{
                  withDockerRegistry(credentialsId: 'docker', toolName: 'docker'){   
                      sh "docker build -t robot-gallery ."
                      sh "docker tag robot-gallery your-dockerhub-username/robot-gallery:tagname "
                      sh "docker push your-dockerhub-username/robot-gallery:tagname "
                     }
                 }
             }
         }
    }
}
